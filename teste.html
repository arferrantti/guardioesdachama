        <ul id="confirmed-list" class="list"></ul>

        <button id="admin-panel" class="admin-btn">Abrir Painel de Administração</button>

    <script>

        // Listas iniciais de convidados

        const initialGuests = [

            "CONVIDADO 1",

            "CONVIDADO 2",

            "CONVIDADO 3",

            "CONVIDADO 4"

        ];

 

        const guestList = document.getElementById("guest-list");

        const confirmedList = document.getElementById("confirmed-list");

        const guestCount = document.getElementById("guest-count");

 

        // Histórico de ações

        let actionHistory = [];

 

        function updateGuestCount() {

            guestCount.textContent = guestList.childElementCount;

        }

 

        function addGuestToList(name, list, action) {

            const listItem = document.createElement("li");

            listItem.textContent = name;

            listItem.classList.add("list-item");

 

            // Clique para mover entre as listas

            listItem.onclick = () => {

                if (list === guestList) {

                    guestList.removeChild(listItem);

                    confirmedList.appendChild(listItem);

                    logAction(name, "Confirmar Presença");

                } else {

                    confirmedList.removeChild(listItem);

                    guestList.appendChild(listItem);

                    logAction(name, "Desconfirmar Presença");

                }

                updateGuestCount();

            };

 

            list.appendChild(listItem);

            updateGuestCount();

 

            if (action) logAction(name, action);

        }

 

        function logAction(name, action) {

            const timestamp = new Date().toLocaleString();

            actionHistory.push(`${timestamp} - ${action}: ${name}`);

        }

 

        // Inicializar as listas

        initialGuests.forEach(name => addGuestToList(name, guestList));

 

        // Ações de administração

        document.getElementById("admin-panel").onclick = () => {

            const adminPage = window.open("", "_blank");

            adminPage.document.write(`

                <h1>Painel de Administração</h1>

                <button onclick="window.close()">Voltar</button>

                <h3>Histórico de Ações</h3>

                <ul>${actionHistory.map(action => `<li>${action}</li>`).join("")}</ul>

            `);

        };

    </script>
